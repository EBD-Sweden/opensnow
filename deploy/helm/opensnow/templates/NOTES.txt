{{- if eq .Values.gateway.type "LoadBalancer" }}

Access the query API:
  export SERVICE_IP=$(kubectl get svc {{ include "opensnow.fullname" . }}-gateway -n {{ .Release.Namespace }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  # HTTP API
  curl http://$SERVICE_IP:{{ .Values.gateway.port }}/health
  # PostgreSQL wire protocol
  {{- if .Values.coordinator.pgwireEnabled }}
  psql -h $SERVICE_IP -p {{ .Values.gateway.postgresPort }} -U opensnow
  {{- else }}
  PostgreSQL wire protocol is disabled by default. Enable only for trusted environments:
    helm upgrade --install {{ .Release.Name }} <chart-ref> --set coordinator.pgwireEnabled=true
  {{- end }}
{{- else if eq .Values.gateway.type "NodePort" }}

Access the query API (NodePort):
  export NODE_IP=$(kubectl get nodes -o jsonpath='{.items[0].status.addresses[?(@.type=="InternalIP")].address}')
  curl http://$NODE_IP:30080/health
  {{- if .Values.coordinator.pgwireEnabled }}
  psql -h $NODE_IP -p 30543 -U opensnow
  {{- else }}
  PostgreSQL wire protocol is disabled by default. Enable only for trusted environments:
    helm upgrade --install {{ .Release.Name }} <chart-ref> --set coordinator.pgwireEnabled=true
  {{- end }}
{{- else }}

Access the query API (port-forward):
  kubectl port-forward svc/{{ include "opensnow.fullname" . }}-gateway {{ .Values.gateway.port }}:{{ .Values.gateway.port }} -n {{ .Release.Namespace }}
  curl http://localhost:{{ .Values.gateway.port }}/health
  {{- if .Values.coordinator.pgwireEnabled }}
  # PostgreSQL wire protocol (trusted opt-in only)
  kubectl port-forward svc/{{ include "opensnow.fullname" . }}-gateway {{ .Values.gateway.postgresPort }}:{{ .Values.gateway.postgresPort }} -n {{ .Release.Namespace }}
  psql -h localhost -p {{ .Values.gateway.postgresPort }} -U opensnow
  {{- else }}
  PostgreSQL wire protocol is disabled by default. Enable only for trusted environments:
    helm upgrade --install {{ .Release.Name }} <chart-ref> --set coordinator.pgwireEnabled=true
  {{- end }}
{{- end }}
