Tables

cdrs - 1000 rows
id, caller, callee, call_type, duration_sec, cost, ts

subscribers - 500 rows
id, phone, name, region, plan, signup_date

transactions - 2000 rows
id, account_id, txn_type, amount, channel, ts

accounts - 500 rows
id, holder_name, account_type, balance, branch, opened_date

Example Queries

SELECT call_type, COUNT(*) as cnt FROM cdrs GROUP BY call_type
SELECT txn_type, ROUND(SUM(amount),2) as total FROM transactions GROUP BY txn_type
SELECT s.region, COUNT(*) as calls FROM subscribers s JOIN cdrs c ON s.phone = c.caller GROUP BY 1 ORDER BY calls DESC
SELECT channel, COUNT(*) as cnt, ROUND(AVG(amount),2) as avg_amt FROM transactions GROUP BY channel
SELECT plan, COUNT(*) as subs, ROUND(AVG(julianday('2026-01-01') - julianday(signup_date)),0) as avg_days FROM subscribers GROUP BY plan
SELECT a.branch, a.account_type, COUNT(t.id) as txns, ROUND(SUM(t.amount),2) as volume FROM accounts a JOIN transactions t ON a.id = t.account_id GROUP BY 1,2 ORDER BY volume DESC LIMIT 15
SELECT substr(ts,1,10) as day, call_type, COUNT(*) as cnt, ROUND(SUM(cost),2) as total_cost FROM cdrs GROUP BY 1,2 ORDER BY 1 LIMIT 20
